# Smartslate Polaris Slidev Integration Guide

Complete guide for integrating the Slidev presentation theme with the Smartslate Polaris platform.

## Overview

This guide shows how to programmatically generate Slidev presentations from blueprint data stored in the `blueprint_generator` table.

---

## Architecture

```
Blueprint Data (Supabase)
  → Markdown Generator (Server-side API)
  → Slidev Theme (Custom layouts)
  → Static HTML Export (Build process)
  → Shareable URL (Public hosting)
```

---

### 2. Create Markdown Generator Service

Create `/frontend/lib/services/slidevGenerationService.ts`:

```typescript
import { createClient } from '@/lib/supabase/server';
import fs from 'fs/promises';
import path from 'path';

interface BlueprintSlide {
  layout: string;
  frontmatter: Record<string, any>;
  content: string;
}

export async function generateSlidevPresentation(blueprintId: string): Promise<string> {
  const supabase = await createClient();

  // Fetch blueprint data
  const { data: blueprint, error } = await supabase
    .from('blueprint_generator')
    .select('*')
    .eq('id', blueprintId)
    .single();

  if (error || !blueprint) {
    throw new Error('Blueprint not found');
  }

  const blueprintJson = blueprint.blueprint_json;
  const slides: BlueprintSlide[] = [];

  // 1. Cover Slide
  slides.push({
    layout: 'cover',
    frontmatter: {
      theme: './theme',
      title: blueprintJson.metadata?.title || 'Learning Blueprint',
      subtitle: blueprintJson.metadata?.description || '',
      author: blueprint.static_answers?.instructorName || 'Smartslate Polaris',
      date: new Date().toLocaleDateString('en-US', {
        year: 'numeric',
        month: 'long',
        day: 'numeric',
      }),
      duration: blueprintJson.metadata?.duration || 'TBD',
      class: 'text-left',
      highlighter: 'shiki',
      lineNumbers: false,
    },
    content: '# Welcome to your Learning Blueprint\n\nGenerated by Smartslate Polaris',
  });

  // 2. Executive Summary Section
  if (blueprintJson.executiveSummary) {
    slides.push({
      layout: 'section',
      frontmatter: {
        layout: 'section',
        sectionNumber: '01',
      },
      content: '# Executive Summary\n\nComprehensive overview of the learning blueprint',
    });

    slides.push({
      layout: 'default',
      frontmatter: {
        title: 'Executive Summary',
      },
      content: formatExecutiveSummary(blueprintJson.executiveSummary),
    });
  }

  // 3. Learning Objectives (Metrics Layout)
  if (blueprintJson.learningObjectives) {
    slides.push({
      layout: 'section',
      frontmatter: {
        layout: 'section',
        sectionNumber: '02',
      },
      content: '# Learning Objectives\n\nWhat participants will achieve',
    });

    slides.push({
      layout: 'metrics',
      frontmatter: {
        layout: 'metrics',
        title: 'Key Learning Metrics',
      },
      content: formatLearningObjectivesAsMetrics(blueprintJson.learningObjectives),
    });
  }

  // 4. Target Audience
  if (blueprintJson.targetAudience) {
    slides.push({
      layout: 'default',
      frontmatter: {
        title: 'Target Audience',
      },
      content: formatTargetAudience(blueprintJson.targetAudience),
    });
  }

  // 5. Content Outline (Timeline)
  if (blueprintJson.contentOutline) {
    slides.push({
      layout: 'section',
      frontmatter: {
        layout: 'section',
        sectionNumber: '03',
      },
      content: '# Content Outline\n\nModule structure and progression',
    });

    slides.push({
      layout: 'timeline',
      frontmatter: {
        layout: 'timeline',
        title: 'Learning Journey',
      },
      content: formatContentOutlineAsTimeline(blueprintJson.contentOutline),
    });
  }

  // 6. Resources & Budget (Table)
  if (blueprintJson.resources) {
    slides.push({
      layout: 'table',
      frontmatter: {
        layout: 'table',
        title: 'Resources & Budget',
      },
      content: formatResourcesAsTable(blueprintJson.resources),
    });
  }

  // 7. Assessment Strategy (Two-Column)
  if (blueprintJson.assessmentStrategy) {
    slides.push({
      layout: 'two-cols',
      frontmatter: {
        layout: 'two-cols',
        title: 'Assessment Strategy',
      },
      content: formatAssessmentTwoColumn(blueprintJson.assessmentStrategy),
    });
  }

  // 8. Implementation Timeline
  if (blueprintJson.implementationTimeline) {
    slides.push({
      layout: 'timeline',
      frontmatter: {
        layout: 'timeline',
        title: 'Implementation Roadmap',
      },
      content: formatImplementationTimeline(blueprintJson.implementationTimeline),
    });
  }

  // 9. Risk Mitigation (Table)
  if (blueprintJson.riskMitigation) {
    slides.push({
      layout: 'table',
      frontmatter: {
        layout: 'table',
        title: 'Risk Mitigation',
      },
      content: formatRisksAsTable(blueprintJson.riskMitigation),
    });
  }

  // 10. Success Metrics
  if (blueprintJson.successMetrics) {
    slides.push({
      layout: 'metrics',
      frontmatter: {
        layout: 'metrics',
        title: 'Success Metrics',
      },
      content: formatSuccessMetrics(blueprintJson.successMetrics),
    });
  }

  // 11. Sustainability Plan
  if (blueprintJson.sustainabilityPlan) {
    slides.push({
      layout: 'default',
      frontmatter: {
        title: 'Sustainability Plan',
      },
      content: formatSustainabilityPlan(blueprintJson.sustainabilityPlan),
    });
  }

  // 12. Closing Slide
  slides.push({
    layout: 'cover',
    frontmatter: {
      layout: 'cover',
    },
    content:
      '# Thank You\n\nQuestions?\n\n<div class="mt-8 opacity-70">\nGenerated by <strong>Smartslate Polaris</strong>\n<br>\nPowered by Solara AI\n</div>',
  });

  // Convert slides to Markdown
  return slidesToMarkdown(slides);
}

// Helper Functions

function slidesToMarkdown(slides: BlueprintSlide[]): string {
  const parts: string[] = [];

  // First slide includes frontmatter
  const firstSlide = slides[0];
  parts.push('---');
  Object.entries(firstSlide.frontmatter).forEach(([key, value]) => {
    if (typeof value === 'string') {
      parts.push(`${key}: '${value}'`);
    } else {
      parts.push(`${key}: ${value}`);
    }
  });
  parts.push('---\n');
  parts.push(firstSlide.content);
  parts.push('\n---\n');

  // Remaining slides
  for (let i = 1; i < slides.length; i++) {
    const slide = slides[i];

    if (Object.keys(slide.frontmatter).length > 0) {
      parts.push('---');
      Object.entries(slide.frontmatter).forEach(([key, value]) => {
        if (typeof value === 'string') {
          parts.push(`${key}: '${value}'`);
        } else {
          parts.push(`${key}: ${value}`);
        }
      });
      parts.push('---\n');
    }

    parts.push(slide.content);

    if (i < slides.length - 1) {
      parts.push('\n---\n');
    }
  }

  return parts.join('\n');
}

function formatExecutiveSummary(summary: any): string {
  return `
**Program Overview**: ${summary.overview || 'Comprehensive learning program'}

**Key Highlights**:
${summary.highlights?.map((h: string) => `- ${h}`).join('\n') || '- Structured learning path'}

**Expected Outcomes**:
${summary.outcomes?.map((o: string, i: number) => `${i + 1}. ${o}`).join('\n') || '1. Enhanced skills and knowledge'}
`;
}

function formatLearningObjectivesAsMetrics(objectives: any[]): string {
  const metrics = objectives.slice(0, 4).map(
    (obj) => `
<div class="metric-card">
  <div class="metric-value">${obj.count || '5'}</div>
  <div class="metric-label">${obj.category || 'Objectives'}</div>
  <div class="metric-subtitle">${obj.description || 'Learning goals'}</div>
</div>
`
  );

  return metrics.join('\n');
}

function formatTargetAudience(audience: any): string {
  return `
<div class="glass-card p-6">

### Primary Audience
**${audience.primary?.title || 'Professionals'}**

${audience.primary?.description || 'Target learners for this program'}

${audience.primary?.personas?.map((p: string) => `- ${p}`).join('\n') || ''}

### Prerequisites
${audience.prerequisites?.map((p: string) => `- ${p}`).join('\n') || '- Basic knowledge required'}

</div>
`;
}

function formatContentOutlineAsTimeline(outline: any[]): string {
  return `
<div class="timeline">
${outline
  .map(
    (module, i) => `
  <div class="timeline-item">
    <h3>${module.title || `Module ${i + 1}`}</h3>
    <p>${module.description || 'Module content description'}</p>
  </div>
`
  )
  .join('\n')}
</div>
`;
}

function formatResourcesAsTable(resources: any): string {
  const rows =
    resources.items
      ?.map(
        (item: any) => `| ${item.name} | ${item.description} | $${item.cost} | ${item.duration} |`
      )
      .join('\n') || '| Example | Description | $1,000 | 1 week |';

  return `
| Resource | Description | Cost (USD) | Duration |
|----------|-------------|------------|----------|
${rows}
`;
}

function formatAssessmentTwoColumn(assessment: any): string {
  return `
::left::

### Formative Assessments
${assessment.formative?.map((a: string) => `- ${a}`).join('\n') || '- Ongoing quizzes'}

**Frequency**: ${assessment.formativeFrequency || 'Weekly'}

::right::

### Summative Assessments
${assessment.summative?.map((a: string) => `- ${a}`).join('\n') || '- Final exam'}

**Frequency**: ${assessment.summativeFrequency || 'End of program'}
`;
}

function formatImplementationTimeline(timeline: any[]): string {
  return formatContentOutlineAsTimeline(timeline);
}

function formatRisksAsTable(risks: any[]): string {
  const rows =
    risks
      .map((risk) => `| ${risk.name} | ${risk.probability} | ${risk.impact} | ${risk.mitigation} |`)
      .join('\n') || '| Example Risk | Medium | High | Mitigation strategy |';

  return `
| Risk | Probability | Impact | Mitigation Strategy |
|------|-------------|--------|---------------------|
${rows}
`;
}

function formatSuccessMetrics(metrics: any[]): string {
  return metrics
    .slice(0, 4)
    .map(
      (metric) => `
<div class="metric-card">
  <div class="metric-value">${metric.target || '90%'}</div>
  <div class="metric-label">${metric.name || 'Success Rate'}</div>
  <div class="metric-subtitle">${metric.description || 'Target metric'}</div>
</div>
`
    )
    .join('\n');
}

function formatSustainabilityPlan(plan: any): string {
  return `
**Long-Term Viability Strategies**:

${
  plan.strategies
    ?.map(
      (s: any, i: number) => `
${i + 1}. **${s.title}**
   ${s.details?.map((d: string) => `- ${d}`).join('\n   ') || ''}
`
    )
    .join('\n') || '1. Continuous improvement process'
}
`;
}
```

### 3. Create API Route for Presentation Generation

Create `/frontend/app/api/blueprints/generate-presentation/route.ts`:

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { createClient } from '@/lib/supabase/server';
import { generateSlidevPresentation } from '@/lib/services/slidevGenerationService';
import fs from 'fs/promises';
import path from 'path';
import { exec } from 'child_process';
import { promisify } from 'util';

const execAsync = promisify(exec);

export async function POST(request: NextRequest) {
  try {
    const supabase = await createClient();

    // Check authentication
    const {
      data: { user },
      error: authError,
    } = await supabase.auth.getUser();
    if (authError || !user) {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
    }

    const { blueprintId } = await request.json();

    if (!blueprintId) {
      return NextResponse.json({ error: 'Blueprint ID required' }, { status: 400 });
    }

    // Verify blueprint ownership
    const { data: blueprint, error } = await supabase
      .from('blueprint_generator')
      .select('id, user_id')
      .eq('id', blueprintId)
      .eq('user_id', user.id)
      .single();

    if (error || !blueprint) {
      return NextResponse.json({ error: 'Blueprint not found' }, { status: 404 });
    }

    // Generate Slidev markdown
    const markdown = await generateSlidevPresentation(blueprintId);

    // Write to file
    const presentationDir = path.join(process.cwd(), 'slidev', 'presentations');
    await fs.mkdir(presentationDir, { recursive: true });

    const markdownPath = path.join(presentationDir, `${blueprintId}.md`);
    await fs.writeFile(markdownPath, markdown, 'utf-8');

    // Build static presentation
    const buildDir = path.join(process.cwd(), 'public', 'presentations', blueprintId);
    await fs.mkdir(buildDir, { recursive: true });

    // Run Slidev build
    await execAsync(
      `npx slidev build ${markdownPath} --base /presentations/${blueprintId}/ --out ${buildDir}`
    );

    // Generate shareable URL
    const presentationUrl = `/presentations/${blueprintId}/index.html`;

    // Update blueprint with presentation URL
    await supabase
      .from('blueprint_generator')
      .update({
        presentation_url: presentationUrl,
        presentation_generated_at: new Date().toISOString(),
      })
      .eq('id', blueprintId);

    return NextResponse.json({
      success: true,
      presentationUrl,
      message: 'Presentation generated successfully',
    });
  } catch (error) {
    console.error('Presentation generation error:', error);
    return NextResponse.json({ error: 'Failed to generate presentation' }, { status: 500 });
  }
}
```

### 5. Create React Hook for Presentation Generation

Create `/frontend/lib/hooks/usePresentationGeneration.ts`:

```typescript
import { useState } from 'react';
import { toast } from 'sonner';

export function usePresentationGeneration() {
  const [isGenerating, setIsGenerating] = useState(false);
  const [presentationUrl, setPresentationUrl] = useState<string | null>(null);

  async function generatePresentation(blueprintId: string) {
    setIsGenerating(true);
    setPresentationUrl(null);

    try {
      const response = await fetch('/api/blueprints/generate-presentation', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ blueprintId }),
      });

      if (!response.ok) {
        throw new Error('Failed to generate presentation');
      }

      const data = await response.json();
      setPresentationUrl(data.presentationUrl);

      toast.success('Presentation generated successfully!');

      return data.presentationUrl;
    } catch (error) {
      console.error('Presentation generation error:', error);
      toast.error('Failed to generate presentation');
      throw error;
    } finally {
      setIsGenerating(false);
    }
  }

  return {
    generatePresentation,
    isGenerating,
    presentationUrl,
  };
}
```

### 6. Add UI Component for Presentation Button

Add to `/frontend/components/features/blueprints/BlueprintActions.tsx`:

```tsx
import { Button } from '@/components/ui/button';
import { usePresentationGeneration } from '@/lib/hooks/usePresentationGeneration';
import { PresentationIcon } from 'lucide-react';

interface PresentationButtonProps {
  blueprintId: string;
}

export function PresentationButton({ blueprintId }: PresentationButtonProps) {
  const { generatePresentation, isGenerating, presentationUrl } = usePresentationGeneration();

  const handleGenerate = async () => {
    const url = await generatePresentation(blueprintId);
    if (url) {
      window.open(url, '_blank');
    }
  };

  return (
    <Button
      onClick={handleGenerate}
      disabled={isGenerating}
      variant="secondary"
      size="large"
      className="min-h-[44px]"
    >
      <PresentationIcon className="mr-2 h-5 w-5" />
      {isGenerating ? 'Generating Presentation...' : 'Generate Slidev Presentation'}
    </Button>
  );
}
```

## Deployment

### Production Build Process

1. **API generates markdown** → `/slidev/presentations/{blueprintId}.md`
2. **Slidev builds static HTML** → `/public/presentations/{blueprintId}/`
3. **Next.js serves static files** → `https://yourapp.com/presentations/{blueprintId}/`

### Vercel Configuration

Add to `vercel.json`:

```json
{
  "buildCommand": "npm run build && npm run slidev:build-all",
  "outputDirectory": ".next",
  "framework": "nextjs",
  "rewrites": [
    {
      "source": "/presentations/:blueprintId/(.*)",
      "destination": "/presentations/:blueprintId/$1"
    }
  ]
}
```
